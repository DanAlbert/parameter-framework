118 {

120 
121  bzero(_aPollFds, sizeof(_aPollFds));
122 
123  // Build poll elements
124  _aPollFds[0].fd = _pListeningSocket->getFd();
125  _aPollFds[1].fd = _aiInbandPipe[0];
126  _aPollFds[0].events = POLLIN;
127  _aPollFds[1].events = POLLIN;
128

133  if (_aPollFds[0].revents & POLLIN) {
